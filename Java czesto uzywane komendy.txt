🔹 1. Czytanie pliku linia po linii
-------------------------------------------------
try (BufferedReader reader = new BufferedReader(new FileReader("plik.csv"))) {
    String line;
    while ((line = reader.readLine()) != null) {
        // Przetwarzanie jednej linii
        System.out.println(line);
    }
} catch (IOException e) {
    e.printStackTrace();
}
